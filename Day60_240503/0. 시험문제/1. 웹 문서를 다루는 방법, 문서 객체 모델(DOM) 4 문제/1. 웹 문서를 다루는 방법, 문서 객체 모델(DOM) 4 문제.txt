1. quiz-1.html 문서를 열어 보면 제목과 텍스트 단락이 있다. 
텍스트 단락을 누르면 글자 크기를 20px로, 
글자색을 blue로 바꾸고 배경색은 #ccc로 지정하는 소스를 작성하자.

//case 1
var myText = document.querySelector("#myText");
myText.addEventListener("click", function()
{
	myText.style.fontSize = "20px";
	myText.style.color = "blue";
	myText.style.backgroundColor = "#ccc";
});

//case 2
var myText = document.querySelector("#myText");
myText.onclick = function()
{
	myText.style.fontSize = "20px";
	myText.style.color = "blue";
	myText.style.backgroundColor = "#ccc";
};

2. quiz-2.html 문서를 열어 보면 다섯 개의 항목이 나열되어 있다. 
각 항목 앞에 있는 체크 표시를 누르면 항목 텍스트의 글자 색이 
회색(#ccc)으로 바뀌도록 소스를 작성하자. -TODO LIST에 접목하면 좋을듯

var buttons = document.querySelectorAll(".check");
for(var i=0; i<buttons.length; i++)
{
	buttons[i].addEventListener("click", function()
	{
		this.parentNode.style.color = "#ccc";
	});
}

document.querySelectorAll(".check");
문서 내에서 클래스 이름이 "check"인 모든 요소를 선택합니다. 
이 메소드는 NodeList를 반환합니다. 
NodeList는 유사 배열 객체로, .length 속성과 배열 인덱스를 사용하여 
각 요소에 접근할 수 있습니다.

buttons[i].addEventListener("click", function()
{
	this.parentNode.style.color = "#ccc";
});
this.parentNode.style.color = "#ccc": 이벤트가 발생한 요소의 
부모 요소(this.parentNode)의 스타일을 변경하여 텍스트 색상을 #ccc로 설정합니다. 
여기서 this는 이벤트가 발생한 요소를 가리킵니다.
